 CREATE TABLE EMP (ENO NUMBER(6) PRIMARY KEY, ENAME VARCHAR2(20) NOT NULL,
2 EBDATE DATE, ADDRESS VARCHAR2(20), GENDER CHAR, SALARY NUMBER(10) NOT NULL,
3 DEPTNO NUMBER(4) REFERENCES DEPT );

 DESC DEPT;

DESC EMP;
